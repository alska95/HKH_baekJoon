class Solution {
    public int maxProfit(int[] prices) {
        int profit = 0;
        for (int i = 0 ; i < prices.length ; i++) {
            int nextI = i + 1;
            if (nextI < prices.length && prices[nextI] > prices[i]) {
                profit += prices[nextI] - prices[i];
            }
        }
        return profit;
    }
}
